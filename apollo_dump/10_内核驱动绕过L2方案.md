# 内核驱动绕过 L2 方案

> 本文档详细说明如何通过内核驱动绕过 L2（FreeStyle.exe 内嵌反调试）的限制。
> 目标：在内核层设置硬件断点，追踪练习场阶段的 ItemCode 读取来源。

---

## 一、技术原理

### 1.1 为什么内核层可以绕过 L2

```
用户态（L2 所在）：
  ├── L2 hook 了 Wow64SetThreadContext
  ├── L2 扫描 .text 页属性
  └── L2 检测调试器附加

内核态（L2 管不到）：
  ├── 可以直接操作进程上下文
  ├── 可以直接读写进程内存
  ├── 可以直接设置调试寄存器
  └── 可以注入 APC 到目标线程
```

**核心原理**：L2 的所有保护机制都在用户态，内核代码执行时 L2 无法感知或拦截。

### 1.2 内核驱动能做什么

| 能力 | 用户态（被 L2 阻止） | 内核态（不受限） |
|------|---------------------|-----------------|
| 设置硬件断点 | ❌ Wow64SetThreadContext 被 hook | ✅ 直接写 DR 寄存器 |
| 读取进程内存 | ✅ ReadProcessMemory 可用 | ✅ 直接读内核映射 |
| 写入进程内存 | ✅ WriteProcessMemory 可用 | ✅ 直接写内核映射 |
| 注入代码 | ❌ 会被检测 | ✅ APC 注入 |
| 捕获断点异常 | ❌ 调试器被阻止 | ✅ 异常回调 |

### 3.2 关键数据结构

```c
// 断点设置请求
typedef struct _SET_BP_REQUEST {
    ULONG ProcessId;         // 目标进程 PID
    ULONG ThreadId;          // 目标线程 TID（0 = 所有线程）
    ULONG64 Address;         // 断点地址
    ULONG Type;              // 断点类型：0=执行, 1=写入, 2=读/写
    ULONG Length;            // 断点长度：1/2/4/8 字节
    ULONG DrIndex;           // DR 索引：0-3
} SET_BP_REQUEST, *PSET_BP_REQUEST;

// 断点触发信息
typedef struct _BP_HIT_INFO {
    ULONG HitCount;          // 触发次数
    ULONG64 TriggerAddress;  // 触发时的 EIP/RIP
    ULONG64 HitAddress;      // 触发断点的地址
    ULONG DrIndex;           // 触发的 DR 索引
    LARGE_INTEGER Timestamp; // 时间戳
} BP_HIT_INFO, *PBP_HIT_INFO;

// 驱动全局状态
typedef struct _DRIVER_GLOBAL {
    PEPROCESS TargetProcess;    // 目标进程对象
    KAPC_STATE ApcState;        // APC 状态
    BP_HIT_INFO HitInfo[100];   // 触发记录
    ULONG HitCount;             // 总触发次数
    KSPIN_LOCK Lock;            // 自旋锁
} DRIVER_GLOBAL, *PDRIVER_GLOBAL;
```

### 3.3 核心代码框架

```c
// DriverEntry
NTSTATUS DriverEntry(PDRIVER_OBJECT DriverObject, PUNICODE_STRING RegistryPath) {
    // 创建设备
    UNICODE_STRING deviceName = RTL_CONSTANT_STRING(L"\\Device\\HwBpDriver");
    IoCreateDevice(DriverObject, 0, &deviceName, FILE_DEVICE_UNKNOWN, 0, FALSE, &deviceObject);
    
    // 创建符号链接
    UNICODE_STRING symLink = RTL_CONSTANT_STRING(L"\\??\\HwBpDriver");
    IoCreateSymbolicLink(&symLink, &deviceName);
    
    // 设置 IRP 处理函数
    DriverObject->MajorFunction[IRP_MJ_CREATE] = HwBpCreate;
    DriverObject->MajorFunction[IRP_MJ_CLOSE] = HwBpClose;
    DriverObject->MajorFunction[IRP_MJ_DEVICE_CONTROL] = HwBpDeviceControl;
    DriverObject->DriverUnload = HwBpUnload;
    
    return STATUS_SUCCESS;
}

// 设置硬件断点
NTSTATUS SetHardwareBreakpoint(PSET_BP_REQUEST Request) {
    NTSTATUS status;
    PEPROCESS process;
    PETHREAD thread;
    CONTEXT context;
    
    // 查找进程
    status = PsLookupProcessByProcessId((HANDLE)Request->ProcessId, &process);
    if (!NT_SUCCESS(status)) return status;
    
    // 附加到进程上下文
    KeStackAttachProcess(process, &g_Global.ApcState);
    
    // 获取线程上下文
    if (Request->ThreadId != 0) {
        PsLookupThreadByThreadId((HANDLE)Request->ThreadId, &thread);
        // 获取线程上下文...
    }
    
    // 设置 DR 寄存器
    // DR0-DR3: 断点地址
    // DR6: 状态寄存器
    // DR7: 控制寄存器
    
    __writegsqword(FS_GS_DR0 + Request->DrIndex * 8, Request->Address);
    
    // 设置 DR7 控制位
    ULONG64 dr7 = __readgsqword(FS_GS_DR7);
    dr7 |= (1 << (Request->DrIndex * 2));  // 局部启用
    dr7 |= ((ULONG64)Request->Type << (16 + Request->DrIndex * 4));
    dr7 |= ((ULONG64)Request->Length << (18 + Request->DrIndex * 4));
    __writegsqword(FS_GS_DR7, dr7);
    
    // 分离进程上下文
    KeUnstackDetachProcess(&g_Global.ApcState);
    
    ObDereferenceObject(process);
    return STATUS_SUCCESS;
}
```

### 3.4 异常捕获

```c
// 注册异常回调
VOID RegisterExceptionCallback() {
    // 方法 1：使用 KeRegisterNmiCallback（需要特定 Windows 版本）
    // 方法 2：使用 IDT hook（高风险）
    // 方法 3：使用内核调试器接口
    
    // 最安全的方法：使用 KiUserExceptionDispatcher 的机制
    // 或者通过 PsSetLoadImageNotifyCallback 监控模块加载
}

// 异常处理
NTSTATUS ExceptionHandler(PEXCEPTION_RECORD ExceptionRecord, 
                          PKTRAP_FRAME TrapFrame) {
    if (ExceptionRecord->ExceptionCode == STATUS_SINGLE_STEP) {
        // 硬件断点触发
        ULONG dr6 = __readgsqword(FS_GS_DR6);
        
        // 记录触发信息
        g_Global.HitInfo[g_Global.HitCount].TriggerAddress = TrapFrame->Rip;
        g_Global.HitInfo[g_Global.HitCount].DrIndex = dr6 & 0xF;
        g_Global.HitCount++;
        
        // 清除 DR6
        __writegsqword(FS_GS_DR6, 0);
        
        return EXCEPTION_CONTINUE_EXECUTION;
    }
    return EXCEPTION_CONTINUE_SEARCH;
}
```

## 四、用户态控制程序

### 4.1 功能

```
HwBpClient.exe
│
├── 打开驱动设备
│   └── CreateFile("\\\\.\\HwBpDriver")
│
├── 设置断点
│   ├── 输入：PID, TID, 地址, 类型
│   └── DeviceIoControl(IOCTL_SET_BP)
│
├── 获取触发记录
│   └── DeviceIoControl(IOCTL_GET_HIT)
│
└── 清除断点
    └── DeviceIoControl(IOCTL_CLEAR_BP)
```

### 4.2 Python 控制脚本

```python
# hwbp_driver_client.py
import ctypes
from ctypes import wintypes

kernel32 = ctypes.windll.kernel32

# IOCTL 定义
IOCTL_SET_BP = 0x800
IOCTL_GET_HIT = 0x804
IOCTL_CLEAR_BP = 0x808

def open_driver():
    return kernel32.CreateFileW(
        r"\\.\HwBpDriver",
        0xC0000000,  # GENERIC_READ | GENERIC_WRITE
        0,
        None,
        3,  # OPEN_EXISTING
        0,
        None
    )

def set_breakpoint(handle, pid, tid, address, dr_index=0, bp_type=2, length=4):
    """设置硬件断点"""
    class SET_BP_REQUEST(ctypes.Structure):
        _fields_ = [
            ("ProcessId", wintypes.ULONG),
            ("ThreadId", wintypes.ULONG),
            ("Address", ctypes.c_uint64),
            ("Type", wintypes.ULONG),
            ("Length", wintypes.ULONG),
            ("DrIndex", wintypes.ULONG),
        ]
    
    req = SET_BP_REQUEST(pid, tid, address, bp_type, length, dr_index)
    bytes_returned = wintypes.DWORD()
    
    return kernel32.DeviceIoControl(
        handle, IOCTL_SET_BP,
        ctypes.byref(req), ctypes.sizeof(req),
        None, 0,
        ctypes.byref(bytes_returned),
        None
    )

def get_hit_info(handle):
    """获取断点触发信息"""
    class BP_HIT_INFO(ctypes.Structure):
        _fields_ = [
            ("HitCount", wintypes.ULONG),
            ("TriggerAddress", ctypes.c_uint64),
            ("HitAddress", ctypes.c_uint64),
            ("DrIndex", wintypes.ULONG),
            ("Timestamp", ctypes.c_uint64),
        ]
    
    info = BP_HIT_INFO()
    bytes_returned = wintypes.DWORD()
    
    kernel32.DeviceIoControl(
        handle, IOCTL_GET_HIT,
        None, 0,
        ctypes.byref(info), ctypes.sizeof(info),
        ctypes.byref(bytes_returned),
        None
    )
    
    return info
```

## 七、风险和注意事项

### 7.1 技术风险

| 风险 | 说明 | 缓解措施 |
|------|------|----------|
| BSOD | 驱动代码错误导致蓝屏 | 充分测试，使用异常处理 |
| 内存访问违规 | 访问无效地址 | 检查地址有效性 |
| 线程同步问题 | 多线程竞争 | 使用自旋锁 |

### 7.2 游戏风险

| 风险 | 说明 |
|------|------|
| 封号 | 内核驱动可能被检测 |
| 游戏更新 | 需要重新适配 |

### 7.3 系统风险

| 风险 | 说明 |
|------|------|
| 测试模式 | 允许未签名驱动，降低系统安全性 |
| 驱动冲突 | 可能与其他驱动冲突 |

## 八、替代方案：使用现有工具

### 8.1 DBK64 / Cheat Engine 驱动

Cheat Engine 自带的 DBK64 驱动已经实现了内核级内存访问和硬件断点功能。

**优点**：
- 已开发完成，直接可用
- 功能完善（内存读写、断点设置）

**缺点**：
- 可能被游戏检测
- 需要安装 CE

### 8.2 WinDbg 内核调试

通过内核调试器可以完全控制目标进程。

**优点**：
- 功能强大
- 官方工具

**缺点**：
- 需要双机调试或本地内核调试
- 操作复杂

### 8.3 Process Hacker / System Informer

可以读写进程内存，但不能设置硬件断点。
